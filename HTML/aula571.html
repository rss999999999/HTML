                public static void bubbleSort(int[] vetor) {
                    int n = vetor.length; // tamanho do vetor
                    for (int k = 0; k < n - 1; k++) { // percorre o vetor n - 1 vezes
                        for (int i = 0; i < n - k - 1; i++) { // compara os elementos adjacentes
                            if (vetor[i] > vetor[i + 1]) { // se o elemento da esquerda for maior que o da direita
                                int aux = vetor[i]; // troca os elementos de posição
                                vetor[i] = vetor[i + 1];
                                vetor[i + 1] = aux;
                            }
                        }
                    }
                }
            }
